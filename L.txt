create or replace FUNCTION fnc_valor_total_pedidos_por_estado(p_estado IN VARCHAR2)
RETURN NUMBER IS
    v_total NUMBER := 0;
BEGIN
    SELECT SUM(hp.val_total_pedido)
    INTO v_total
    FROM historico_pedido hp
    JOIN endereco_cliente ec ON ec.seq_endereco_cliente = hp.seq_endereco_cliente
    JOIN cidade ci ON ci.cod_cidade = ec.cod_cidade
    JOIN estado e ON e.cod_estado = ci.cod_estado
    WHERE UPPER(e.nom_estado) = UPPER(p_estado) AND hp.dat_entrega IS NOT NULL;

    RETURN v_total;
EXCEPTION
    WHEN NO_DATA_FOUND THEN
        RETURN 0;
    WHEN OTHERS THEN
        RETURN -1;
END;
